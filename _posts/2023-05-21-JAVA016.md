	public class basic050 {
		public static String shifts(int a) {
			String s="";
			for(int i=0; i<=31; i++) {
				int b = a%2;
				s = (b>=0)?b+s : (-b)+s;
				a>>=1;
			}
			return s;	
		}
>

		public static void main(String []args) {
			int intNums1 = 123;
			int intNums2 = -123;
			
			System.out.printf("%d : %s %n", intNums1,shifts(intNums1));
			System.out.printf("%d : %s %n", intNums2,shifts(intNums2));
		}
	}
