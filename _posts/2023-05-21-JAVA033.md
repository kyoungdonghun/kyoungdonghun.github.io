public class basic070 {
	public static void main(String[]args) {
		StringBuilder sbu = new	StringBuilder();
		sbu.append("가나")
			.append("다라")
			.append("마바");
		System.out.println(sbu);
		System.out.println("------");
		sbu.replace(1, 2, "붥");
		System.out.println(sbu);
		System.out.println("------");
		sbu.reverse();
		System.out.println(sbu);
		System.out.println("------");
		sbu.deleteCharAt(2);
		System.out.println(sbu);
		System.out.println("------");
		sbu.delete(0, 1);
		System.out.println(sbu);
		System.out.println("------");
		String ss = sbu.substring(0);
		System.out.println(ss);
		System.out.println(sbu);
		System.out.println("------");
		String st = sbu.substring(1);
		System.out.println(st);
		System.out.println(sbu);
		System.out.println("------");
	}
}
