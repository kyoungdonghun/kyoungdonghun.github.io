	public class basic043 {
		public static void showOddnEvenw(int n) {
			int temp = n;
			while(temp != 1) {	
>1이 될때 까지 실행


				if (temp %2 == 1) { 
>나머지가1일때.

					temp=temp*3+1;	
>3을 곱하고 1을 더한다.

				}
				else {
					temp/=2;	
>temp = temp/2

				}
				System.out.println("["+temp+"]");
			}
			System.out.println("\n--------------");
		}
		public static void main(String [] args) {
			showOddnEvenw(10);
		}
	}
